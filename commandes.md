# Charger une branche
git checkout master

# Nouvelle branche copie de l'actuelle
git checkout -b branch_name

# Commit modifs
git commit

# Status
git status

# Merge data depuis master
git pull origin master

# Afficher les branches locales
git branch

# Supprimer une ancienne branche local
git branch -d branch_name

vim save : :wq

vim quit : :q!
